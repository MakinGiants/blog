--- title: How to use NSMutableDictionary example layout: post date: 2012-11-15 00:15:00 UTC updated: 2015-02-08 19:42:31 UTC comments: false categories: IOS --- Example: 

 NSMutableDictionary *dict = [[NSMutableDictionary alloc] init];  
 [dict setObject:@"John" forKey:@"Firstname"];  
 [dict setObject:@"Doe" forKey:@"Lastname"];  
 [dict setObject:@"info at objectgraph.com" forKey:@"Email"];  

 NSLog(@"%@", dict);  

 // Print each key and value  
 NSArray *keys = [dict allKeys];  
 for (NSString *key in keys) {  
    NSLog(@"%@ is %@",key, [dict objectForKey:key]);  
 }
